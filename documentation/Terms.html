Size of a term

Fixpoint size_term (t : term) {struct t} : nat :=
 match t with
  | [a]^t1 => 1 + size_term t1
  | <|t1,t2|> => 1 + size_term t1 + size_term t2
  | Fc m n t1 => 1 + size_term t1
  | _ => 1
 end.
